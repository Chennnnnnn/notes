将大整数用字符串表示，分割成数组，再逐位相加

```
function sumStrings(a,b){
    var res='', c=0;
    a = a.split('');
    b = b.split('');
    while (a.length || b.length || c){
        c +=  ~~a.pop() + ~~b.pop();
        res = c % 10 + res;
        c = c>9;
    }
    console.log(res)
    return res.replace(/^0+/,'');
}

console.log(sumStrings('0002130033','000234234'));
```
