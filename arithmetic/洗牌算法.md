### 洗牌算法

随机交换数组内元素

```
if(!Array.prototype.shuffle){
    Array.prototype.shuffle = function (n) {        
        var len = this.length,
            num = n?Math.min(n,len) : len,
            arr = this.slice(0),
            temp,
            index;
        var range = function (min,max) {
            return min + Math.floor(Math.random()*(max-min+1));
        }
        for (var i =0 ; i < len ; i++){
            index = range(i, len-1);
            temp = arr[i];
            arr[i] = arr[index];
            arr[index] = temp;
        }
        return arr.slice(0, num);
    }
}

随机从原数组中抽取一个元素，加到新数组中

```

if(!Array.prototype.shuffle){
    Array.prototype.shuffle = function (n) {
        var range = function (min,max) {
            return min + Math.floor(Math.random()*(max-min+1));
        }
        var len = this.length,
            num = n?Math.min(n,len) : len,
            arr = this.slice(0),
            result = [],
            index;
        for (var i =0 ; i < num ; i++){
            index = range(i, len-1-i);
            result.push(arr.splice(index,1)[0]);
        }
        return result;
    }
}
