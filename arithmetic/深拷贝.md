### 深拷贝

不考虑循环引用的情况

```
var deepCopy = (obj) => {
    // 判断obj类型,undefined,null,基本类型
    if(obj == null || (typeof obj) !== 'object'){
        return obj;
    }
    //判断是数组还是对象
    let result = Array.isArray(obj)? []: {};
    let objs = [];
    let time = 0;
    for(let key in obj){
        if(obj.hasOwnProperty(key)){
            let temp = obj[key];
            //判断是对象还是基本值
            if(typeof temp === 'object' && temp !== null){   
                if(objs.indexOf(obj[key]) < 0) {
                    if(time > 9){
                        return result;
                    }
                    console.log(time++);
                    objs.push(obj[key]);
                    result[key] = deepCopy(obj[key]);
                    
                } else {
                    result[key] = {};
                }
            } else {
                result[key] = obj[key];
            }
        }
    }
    return result;
}
```
