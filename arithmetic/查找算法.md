### 折半查找

```
function Binary_Search (arr,key) {
    var low,high,mid;
    low = 0;
    high = arr.length -1 ;
    while(low <= high ){
        mid = Math.floor((low + high) /2);
        console.log(mid)
        if (key < arr[mid]) {
            high = mid -1;
        } else if (key > arr[mid]) {
            low = mid + 1;
        } else {
            return mid;
        }
    }
    return -1;
}
```
