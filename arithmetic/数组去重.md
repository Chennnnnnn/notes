##### 使用Set

```
function arrayUnique(arr) {
  return Array.from(new Set(arr))
}

```

##### 使用Map

```
function arrayUnique(arr) {
  const b = new Map();
  return arr.filter((a) => !b.has(a) && b.set(a,1))
}

```

##### 使用indexOf
```
function arrayUnique(arr) {
    var len = arr.length;
    var _arr = [];
    for (let i = 0; i < len; i++) {
        if (_arr.indexOf(arr[i]) === -1) {
            _arr.push(arr[i]);
        }
    }
    return _arr;
}


```
##### 哈希法

```

function arrayUnique (array) {
	var map = {},
		result = [],
		i,
		len = array.length;
	for (i = 0; i < len; i++) {
		if (!map[array[i]]) {
			map[array[i]] = true;
			result.push(array[i]);
		}
	}
	return result;
}
```
