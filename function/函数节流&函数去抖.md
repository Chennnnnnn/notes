#### 函数节流

```
var throttle = function (fn,time){
    var pre = null;
    var t = new Date().getTime();
    var start = true;
    return function () {
        var context = this,
            args = arguments;
        clearTimeout(pre);
        if(start) {
            fn.apply(context,args);
            start = false;
        } else {
            pre = setTimeout(function(){
                fn.apply(context,args);
            },time-(new Date().getTime()-t))
        }       
        t = new Date().getTime();
    }
}

```

#### 函数去抖

```

function debounce(fn, time) {
    var pre = null;
    return function () {
        var context = this,
            args = arguments;
        clearTimeout(pre);
        pre = setTimeout(function () {
            fn.apply(context, args);
        },time)
    }
}

```
